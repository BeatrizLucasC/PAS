SELECT
  ano,
  id_utilizador,
  dominio,
  pontuacao_user AS pontuacao,
  DENSE_RANK() OVER (
    PARTITION BY ano, dominio
    ORDER BY pontuacao_user DESC
  ) AS posicao
FROM vw_score_user_avg_dominio
WHERE dominio = 'Social'
	AND ano = {{Select_year.selectedOptionValue}}
ORDER BY ano DESC, pontuacao DESC, id_utilizador ASC;
